cmake_minimum_required(VERSION 2.6)
project(enet C)

set(ENET_STATIC "0" CACHE BOOL "Create a static library")
set(ENET_SHARED "1" CACHE BOOL "Create a shared library")
set(ENET_LZ4 "0" CACHE BOOL "Enable packet level compression")

if (MSVC)
	add_definitions(-W3)
else()
	add_definitions(-Wno-error)
endif()

include_directories(${PROJECT_SOURCE_DIR})

if (ENET_STATIC)
    if (NOT ENET_LZ4)
        add_library(enet_static STATIC library.c)
    else()
        add_definitions(ENET_LZ4)
        add_library(enet_static STATIC library.c lz4/lz4.c)
    endif()

    if (WIN32)
        target_link_libraries(enet_static winmm ws2_32)
    endif()
endif()

if (ENET_SHARED)
    if (NOT ENET_LZ4)
        add_library(enet SHARED library.c)
    else()
        add_definitions(ENET_LZ4)
        add_library(enet SHARED library.c lz4/lz4.c)
    endif()

    if (WIN32)
        target_link_libraries(enet winmm ws2_32)
    endif()
endif()
